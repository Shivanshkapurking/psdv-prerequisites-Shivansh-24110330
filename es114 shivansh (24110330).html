def sets_(l1,l2):
    results=[]
    l=l1+l2
    results.append(set(l))
    s2,s3=[],[]
    for i in l1:
        if i in l2:
            s2.append(i)
        else:
            s3.append(i)
    results.extend([set(s2),set(s3)])
    return results
